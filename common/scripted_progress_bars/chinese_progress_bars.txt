cfd_modernization_progress_bar = {
	name = "cfd_modernization_progress_bar"
 	desc = "cfd_modernization_progress_bar_desc"
	start_value = 0
 	min_value = 0
 	max_value = 100
	default = yes
 	monthly_progress = {
		add = {
			desc = from_legitimacy
			add = {
				value = this.government_legitimacy
				divide = 60
			}
		}
		add = {
			desc = intelligentsia_clout
			add = {
				ig:ig_intelligentsia = {
					value = this.ig_clout
					multiply = 10
					if = {
						limit = {
							is_in_government = yes
						}
						multiply = 1.5
					}
				}
			}
		}
 	}
}
