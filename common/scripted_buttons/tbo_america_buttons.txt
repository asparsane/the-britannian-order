por_sabotage_barracks = {
	name = "por_sabotage_barracks"
	desc = "por_sabotage_barracks_desc"

	visible = {
		always = yes
	}

	ai_chance = {
		value = 15
		add = {
			value = var:skibidi_postwar_debt_amount
			multiply = -1
		}
	}

	possible = {
		var:skibidi_postwar_debt_amount <= 15
	}

	effect = {
		change_variable = {
			name = barracks_sabotage
			add = 1
		}
		change_variable = {
			name = skibidi_postwar_debt_amount
			add = 1
		}
	}
}

bob_acknowledge_tralaleroism = {
	name = "bob_acknowledge_tralaleroism"
	desc = "bob_acknowledge_tralaleroism_desc"
	visible = {
		AND = {
			has_variable = brainrot_level
			var:brainrot_level >= 3
		}
		NOT = {
			country_has_primary_culture = cu:tralalero
		}
	}
	possible = {
		NOT = {
			country_has_primary_culture = cu:tralalero
		}
	}
	ai_chance = {
		value = 5
		if = {
			limit = {
				AND = {
					has_variable = brainrot_max_tt
					var:brainrot_max_tt >= 2
					has_variable = tralalero_var
				}
			}
			add = {
				value = 100
				subtract = var:tralalero_var
				multiply = 0.05
			}
		}
	}
	effect = {
		add_primary_culture = cu:tralalero
		every_scope_state = {
			convert_population = {
				target = rel:tralaleroist
				value = 0.1
			}
		}
		every_scope_pop = {
			limit = {
				religion = rel:tralaleroist
			}
			change_pop_culture = {
				target = cu:tralalero
				value = 0.2
			}
		}
	}
}
