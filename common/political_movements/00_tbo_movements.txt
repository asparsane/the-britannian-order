movement_fengtian_civicalist = { # fengtian civicalist movement
	category = movement_category_ideological
	
	ideology = ideology_tbo_liberal
	character_ideologies = {
		ideology_fengtian_civicalist
	}
	
	creation_trigger = {
		c:FGT ?= this
	}	
	creation_weight = {
		value = 100
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	
	character_support_trigger = {
		OR = {
			is_interest_group_type = ig_trade_unions
			is_interest_group_type = ig_rural_folk
			is_interest_group_type = ig_intelligentsia
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_fengtian_civicalist
			}
			multiply = {
				value = 10
			}
		}
		if = { # unions extra inclined
			limit = {
				is_interest_group_type = ig_trade_unions
			}
			multiply = {
				value = 2
			}
		}
	}
	
	pop_support_trigger = {
		NOT = {
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_below_expected_sol
		movement_support_engineers
		movement_support_academics
		movement_support_laborers
		movement_support_machinists
		movement_support_clerks
		movement_support_peasants	
	}
	
	pop_support_weight = {	
		add = {
			desc = "URBAN_STATE"
			value = 15
			multiply = state.state_urbanization_rate
		}	

		if = {
			limit = {
				strata = upper
			}
			multiply = {
				value = 0.05
				desc = "UPPER_NO_ICON"
			}
		}	
		else_if = {
			limit = {
				strata = middle
			}
			multiply = {
				value = 2
				desc = "MIDDLE_NO_ICON"
			}
			if = {
				limit = {
					is_pop_type = engineers
				}
				add = {
					value = 10
					desc = "POP_ENGINEERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 20
					desc = "POP_ACADEMICS"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = laborers
				}
				add = {
					value = 18
					desc = "POP_LABORERS"
				}
			}		
			else_if = {
				limit = {
					is_pop_type = machinists
				}
				add = {
					value = 20
					desc = "POP_MACHINISTS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 25
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = peasants 
				}
				add = {
					value = 10
					desc = "POP_PEASANTS"
				}
			}
			else = {
				add = {
					value = 10
					desc = "LOWER_NO_ICON"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol # It is intentional that a labourer being well-off doesn't decrease support for this.
			}
			add = { # Increases likelihood of support if impoverished.
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = state_urbanization_rate
		}
	}
	
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 1000
		}

		state_weight = {
			value = state_urbanization_rate
		}		
	}	
	
	additional_radicalism_factors = {
		owner = {		
		}
	}	
}
