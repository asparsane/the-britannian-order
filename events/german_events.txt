	placement = ROOT
	event_image = {

	}

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = german_events.1.t
	desc = german_events.1.d
	flavor = german_events.1.f

	duration = 3

	trigger = {

	immediate = {
		c:GBR ?= {
			save_scope_as = gbr_scope
		}
		change_relations = {
			country = scope:gbr_scope
			value = -30
		}
		this.ruler = {
			save_scope_as = ruler_scope
		}
	}

	option = {
		name = german_events.1.a
		ai_chance = {
			base = 10
		}
		add_modifier = {
			name = ger_political_chaos_modifier
			days = short_modifier_time
		}
		this.ruler = {
			kill_character = yes
		}
		trigger_event = {
			id = bormann_events.1
			days = 1
			popup = yes
		}
	}
