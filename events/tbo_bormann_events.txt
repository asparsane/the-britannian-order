		}
	}
}

	}

	icon = "gfx/interface/icons/event_icons/event_skull.dds"
	duration = 3

	title = bormann_events.2.t
	desc = bormann_events.2.d
	flavor = bormann_events.2.f

	immediate = {
		random_scope_character = {
			limit = {
				has_role = ig_leader
			}
			save_scope_as = opposer_leader
		}
		random_scope_character = {
			limit = {
				has_template = ger_schnitzel_bormann
			}
			save_scope_as = bormann_scope
		}
	}
	trigger = {
	}
	option ={
		name = bormann_events.2.a
		ai_chance = {
			base = 10
		}
		scope:opposer_leader = {
			kill_character = yes
		}
		scope:bormann_scope = {
			add_modifier = {
				name = bormann_suspected_of_killing_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
	}
}
